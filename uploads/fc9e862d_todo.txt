 select a.MAINACCOUNTID,left(c.SUBLEDGERVOUCHER,6),sum(b.ACCOUNTINGCURRENCYAMOUNT * -1),'0' ISCREDIT
from dbo.MAINACCOUNT a,
dbo.GENERALJOURNALACCOUNTENTRY b,
dbo.GENERALJOURNALENTRY c,
dbo.FISCALCALENDARPERIOD d
where a.ACCOUNTCATEGORYREF <> 0
and a.MAINACCOUNTID = b.LEDGERACCOUNT
and b.GENERALJOURNALENTRY = c.RECID
and c.SUBLEDGERVOUCHERDATAAREAID = 'ga'
and c.ACCOUNTINGDATE >= '20240101'
and c.ACCOUNTINGDATE <= '20240131'
and a.MAINACCOUNTID like '2101-02%'
and c.FISCALCALENDARPERIOD = d.RECID
and d.[TYPE] = '1'
and b.ISCREDIT = 0
and c.POSTINGLAYER = '0'
 group by a.MAINACCOUNTID ,left(c.SUBLEDGERVOUCHER,6)
union all
  select a.MAINACCOUNTID,left(c.SUBLEDGERVOUCHER,6),sum(b.ACCOUNTINGCURRENCYAMOUNT * -1),'1' ISCREDIT
from dbo.MAINACCOUNT a,
dbo.GENERALJOURNALACCOUNTENTRY b,
dbo.GENERALJOURNALENTRY c,
dbo.FISCALCALENDARPERIOD d
where a.ACCOUNTCATEGORYREF <> 0
and a.MAINACCOUNTID = b.LEDGERACCOUNT
and b.GENERALJOURNALENTRY = c.RECID
and c.SUBLEDGERVOUCHERDATAAREAID = 'ga'
and c.ACCOUNTINGDATE >= '20240101'
and c.ACCOUNTINGDATE <= '20240131'
and a.MAINACCOUNTID like '2101-02%'
and c.FISCALCALENDARPERIOD = d.RECID
and d.[TYPE] = '1'
and b.ISCREDIT = 1
and c.POSTINGLAYER = '0'
 group by a.MAINACCOUNTID ,left(c.SUBLEDGERVOUCHER,6)
